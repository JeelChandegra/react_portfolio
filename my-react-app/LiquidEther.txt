--- LiquidEther.tsx ---
import React, { useRef, useEffect } from 'react';
import './LiquidEther.css';

interface LiquidEtherProps {
  width?: number;
  height?: number;
  isViscous?: boolean;
  BFECC?: boolean;
}

const LiquidEther: React.FC<LiquidEtherProps> = ({
  width = window.innerWidth,
  height = window.innerHeight,
  isViscous = false,
  BFECC = true,
}) => {
  const canvasRef = useRef<HTMLCanvasElement | null>(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    canvas.width = width;
    canvas.height = height;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    let frame = 0;
    const animate = () => {
      frame++;
      ctx.clearRect(0, 0, width, height);

      ctx.fillStyle = '#0a0f2f';
      ctx.fillRect(0, 0, width, height);

      ctx.beginPath();
      for (let x = 0; x < width; x += 10) {
        const y =
          height / 2 +
          Math.sin((x + frame * 2) * 0.02) * 50 +
          Math.cos((x + frame * 3) * 0.015) * 30;
        if (x === 0) ctx.moveTo(x, y);
        else ctx.lineTo(x, y);
      }
      ctx.lineTo(width, height);
      ctx.lineTo(0, height);
      ctx.closePath();

      const grad = ctx.createLinearGradient(0, 0, 0, height);
      grad.addColorStop(0, 'rgba(20, 150, 255, 0.4)');
      grad.addColorStop(1, 'rgba(50, 0, 150, 0.6)');
      ctx.fillStyle = grad;
      ctx.fill();

      requestAnimationFrame(animate);
    };

    animate();
  }, [width, height, isViscous, BFECC]);

  return <canvas ref={canvasRef} className="liquid-ether-canvas" />;
};

export default LiquidEther;

.liquid-ether-canvas {

  left: 0;
  z-index: -1;
}
